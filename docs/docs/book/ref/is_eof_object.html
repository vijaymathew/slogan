is_eof_object


is_eof_object(obj)

Return true if obj is the end-of-file object.

Examples:


let f = file_reader("README.md")
letfn loop (c = read_char(f), cs = [])
  if (is_eof_object(c)) reverse(cs)
  else loop(read_char(f), c:cs)
// [a list of characters...]
close_reader(f)
